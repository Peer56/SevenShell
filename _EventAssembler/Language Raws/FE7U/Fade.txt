# Fades the screen out of the opening darkness set in the Chapter Data Table.
# This is mandatory if a chapter has opening darkness. Not using OPEN or using UNFADE_BLACK instead will lead to visual glitches.
OPEN, 0x0C, 4, -game:FE7 -indexMode:8

# Fades the screen out to black. The duration determines how many frames the fade takes to end.
FADE_BLACK, 0x8E, 4, -game:FE7 -indexMode:8
	duration, 2, 2, -preferredBase:10

# Fades the screen in from black to normal. The duration determines how many frames the fade takes to end.
UNFADE_BLACK, 0x8F, 4, -game:FE7 -indexMode:8
	duration, 2, 2, -preferredBase:10

# Fades the screen out to white. The duration determines how many frames the fade takes to end.
FADE_WHITE, 0x90, 4, -game:FE7 -indexMode:8
	duration, 2, 2, -preferredBase:10

# Fades the screen in from white to normal. The duration determines how many frames the fade takes to end.
UNFADE_WHITE, 0x91, 4, -game:FE7 -indexMode:8
	duration, 2, 2, -preferredBase:10

# Fades the screen to a CG. The duration determines how many frames the fade takes to end.
# In a typical fade to a background, you would use FADE_BLACK or FADE_WHITE. In a fade to a CG, this should be used instead.
FADE_CG, 0xD5, 4, -game:FE7 -indexMode:8
	duration, 2, 2, -preferredBase:10

# Fades the screen from a CG. The duration determines how many frames the fade takes to end.
# In a typical fade from a background, you would use UNFADE_BLACK or UNFADE_WHITE. In a fade to a CG, this should be used instead.
UNFADE_CG, 0xD6, 4, -game:FE7 -indexMode:8
	duration, 2, 2, -preferredBase:10
